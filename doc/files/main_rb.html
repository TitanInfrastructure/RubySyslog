Usage: 

  #!/usr/bin/env ruby
  require 'parser'
  require 'storage'
  require 'server'
  require 'yaml'

  config = YAML.load_file('config.yaml')

  server = RubySyslog::Server.new(config)
  server.start

  while !server.stopped? or server.logs_queued? do
    sleep(1)
    server.process_logs
  end

  server.shutdown
