    def assemble
      unless @hostname and @facility and @severity
        return "I AM A JUNK PACKET CUZ MY USER DIDNT SET ME"
      end
      data = "<#{pri}>#{generate_timestamp} #{@hostname} #{@msg}"
      while data.bytesize > 1024
        data = data[0..(data.length-2)]
      end
      data
    end
