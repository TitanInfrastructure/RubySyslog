        def store_log(log)
                new_log = Log.new(
                        :hostname    => log.hostname,
                        :msg         => log.msg,
                        :facility    => log.facility,
                        :severity    => log.severity
                )
                begin
                        new_log.save
                rescue
                        puts "Saving log #{log.object_id.to_s(16)} failed. Retrying... (try tweaking the storage pool size)"
                        sleep(1)
                        retry
                end
        end
